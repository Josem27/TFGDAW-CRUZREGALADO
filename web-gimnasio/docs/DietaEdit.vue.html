export default {
  data() {
    return {
      /**
       * Días de la semana para la dieta.
       * @type {Array<string>}
       */
      dias: ['Lunes', 'Martes', 'Miércoles', 'Jueves', 'Viernes', 'Sábado', 'Domingo'],

      /**
       * Información general de la dieta.
       * @type {Object}
       * @property {string} nombre_dieta - El nombre de la dieta.
       * @property {string} descripcion - Descripción de la dieta.
       * @property {string} fecha_inicio - Fecha de inicio de la dieta.
       * @property {string} fecha_fin - Fecha de fin de la dieta.
       */
      dieta: {
        nombre_dieta: '',
        descripcion: '',
        fecha_inicio: '',
        fecha_fin: '',
      },

      /**
       * Lista de alimentos disponibles por tipo.
       * @type {Array<Object>}
       * @property {number} id_alimento - ID único del alimento.
       * @property {string} nombre_alimento - Nombre del alimento.
       */
      alimentosPorTipo: [
        { id_alimento: 1, nombre_alimento: 'Pollo' },
        { id_alimento: 2, nombre_alimento: 'Arroz' },
      ],

      /**
       * Alimentos asignados para cada día de la semana.
       * @type {Object}
       * @property {Array<Object>} Lunes - Lista de alimentos para el lunes.
       * @property {Array<Object>} Martes - Lista de alimentos para el martes.
       * @property {Array<Object>} Miércoles - Lista de alimentos para el miércoles.
       * @property {Array<Object>} Jueves - Lista de alimentos para el jueves.
       * @property {Array<Object>} Viernes - Lista de alimentos para el viernes.
       * @property {Array<Object>} Sábado - Lista de alimentos para el sábado.
       * @property {Array<Object>} Domingo - Lista de alimentos para el domingo.
       */
      alimentosPorDia: {
        Lunes: [],
        Martes: [],
        Miércoles: [],
        Jueves: [],
        Viernes: [],
        Sábado: [],
        Domingo: [],
      },
    };
  },
  methods: {
    /**
     * Añade una nueva fila de alimento para el día especificado.
     * 
     * @param {string} dia - Día de la semana al que se añadirá el alimento.
     * @returns {void}
     */
    agregarFila(dia) {
      this.alimentosPorDia[dia].push({
        id_alimento: '',
        cantidad: '',
        tiempo_comida: 'desayuno',
      });
    },

    /**
     * Elimina una fila de alimento para el día especificado.
     * 
     * @param {string} dia - Día de la semana del cual se eliminará el alimento.
     * @param {number} index - Índice del alimento a eliminar.
     * @returns {void}
     */
    eliminarFila(dia, index) {
      this.alimentosPorDia[dia].splice(index, 1);
    },

    /**
     * Guarda los cambios realizados en la dieta.
     * 
     * Este método imprime la dieta y los alimentos organizados por día
     * en la consola para depuración. 
     * 
     * @returns {dieta}
     */
    guardarCambios() {
      console.log(this.dieta);
      console.log(this.alimentosPorDia);
    },
  },
};
